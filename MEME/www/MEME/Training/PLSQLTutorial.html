JDBC [5 min]

import java.sql.*;
import java.io.*;

public class Extractor {

  public Extractor() {
    super();
    Extract();
  }

  public static void Extract() {

    Class.forName("oracle.jdbc.driver.OracleDriver");
    Connection conn = DriverManager.getConnection (
        "jdbc:oracle:thin:scott/tiger@pc16:1521:ORCL");

    try {
      Statement stmt = conn.createStatement();

      ResultSet rset =
          stmt.executeQuery("select table_name,max_id from max_tab");

      while (rset.next ()) {
       System.out.print(rset.getString("TABLE_NAME")+"|");
       System.out.print(rset.getString(2)+"\n");
       }
    }
    catch (Exception e) {
      System.out.print("Failed SQL Execution: "+e.getMessage());
 }
}
}
